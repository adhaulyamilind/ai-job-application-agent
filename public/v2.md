### **Inference Algorithm**

```typescript
function inferSkills(
  explicitSkills: string[],
  graph: SkillGraph
): InferredSkill[] {
  const inferred = new Map<string, number>();
  
  for (const skill of explicitSkills) {
    const node = graph.get(skill);
    if (!node) continue;
    
    for (const edge of node.edges) {
      const existing = inferred.get(edge.target) || 0;
      // Take max confidence across paths
      inferred.set(edge.target, Math.max(existing, edge.weight));
    }
  }
  
  return Array.from(inferred.entries()).map(([skill, confidence]) => ({
    skill,
    confidence,
    source: 'inferred'
  }));
}
```

function boostWithEvidence(

  const BOOST_AMOUNT = 0.15;
